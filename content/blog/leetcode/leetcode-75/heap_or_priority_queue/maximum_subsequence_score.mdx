## Problem Statement

You are given two 0-indexed integer arrays `nums1` and `nums2` of equal length `n` and a positive integer `k`. You must choose a subsequence of indices from `nums1` of length `k`.

For chosen indices `i0, i1, ..., ik - 1`, your score is defined as:

- The sum of the selected elements from `nums1` multiplied with the minimum of the selected elements from `nums2`.
- It can be defined simply as: `(nums1[i0] + nums1[i1] + ... + nums1[ik - 1]) * min(nums2[i0], nums2[i1], ... ,nums2[ik - 1])`.

### Examples:

**Example 1:**

```
Input: nums1 = [1,3,3,2], nums2 = [2,1,3,4], k = 3
Output: 12
```

**Example 2:**

```
Input: nums1 = [4,2,3,1,1], nums2 = [7,5,10,9,6], k = 1
Output: 30
```

### Constraints:

- `n == nums1.length == nums2.length`
- `1 <= n <= 10^5`
- `0 <= nums1[i], nums2[j] <= 10^5`
- `1 <= k <= n`

### Implementation in Go

```go
import (
    "container/heap"
    "sort"
)

type Pair struct {
    num1 int
    num2 int
}

type MinHeap []int

func (h MinHeap) Len() int            { return len(h) }
func (h MinHeap) Less(i, j int) bool  { return h[i] < h[j] }
func (h MinHeap) Swap(i, j int)       { h[i], h[j] = h[j], h[i] }
func (h *MinHeap) Push(x interface{}) { *h = append(*h, x.(int)) }
func (h *MinHeap) Pop() interface{} {
    old := *h
    n := len(old)
    x := old[n-1]
    *h = old[:n-1]
    return x
}

func maxScore(nums1 []int, nums2 []int, k int) int64 {
    pairs := make([]Pair, len(nums1))
    for i := range nums1 {
        pairs[i] = Pair{nums1[i], nums2[i]}
    }
    
    sort.Slice(pairs, func(i, j int) bool {
        return pairs[i].num2 > pairs[j].num2
    })
    
    minHeap := &MinHeap{}
    heap.Init(minHeap)
    
    sum, result := 0, 0
    for i, pair := range pairs {
        sum += pair.num1
        heap.Push(minHeap, pair.num1)
        if minHeap.Len() > k {
            sum -= heap.Pop(minHeap).(int)
        }
        if minHeap.Len() == k {
            result = max(result, sum*pair.num2)
        }
    }
    return int64(result)
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
```

### Implementation in Java

```java
import java.util.*;

class Solution {
    public int maxScore(int[] nums1, int[] nums2, int k) {
        int n = nums1.length;
        List<int[]> pairs = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            pairs.add(new int[]{nums1[i], nums2[i]});
        }
        
        pairs.sort((a, b) -> b[1] - a[1]);
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        
        int sum = 0, result = 0;
        for (int[] pair : pairs) {
            sum += pair[0];
            minHeap.offer(pair[0]);
            if (minHeap.size() > k) {
                sum -= minHeap.poll();
            }
            if (minHeap.size() == k) {
                result = Math.max(result, sum * pair[1]);
            }
        }
        return result;
    }
}
```

### Implementation in Python

```python
import heapq

class Solution:
    def maxScore(self, nums1: list[int], nums2: list[int], k: int) -> int:
        n = len(nums1)
        pairs = [(nums1[i], nums2[i]) for i in range(n)]
        
        pairs.sort(key=lambda x: x[1], reverse=True)
        
        min_heap = []
        sum_values = 0
        result = 0
        
        for pair in pairs:
            sum_values += pair[0]
            heapq.heappush(min_heap, pair[0])
            if len(min_heap) > k:
                sum_values -= heapq.heappop(min_heap)
            if len(min_heap) == k:
                result = max(result, sum_values * pair[1])
        
        return result

```



## Conclusion

This problem can be solved efficiently using a **Min Heap (Priority Queue)** to maintain the `k` largest `nums1` values. The time complexity is **O(n log k)** due to heap operations.
