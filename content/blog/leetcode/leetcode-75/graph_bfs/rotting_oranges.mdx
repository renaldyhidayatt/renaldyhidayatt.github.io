## Problem Statement


You are given an `m x n` grid where each cell can have one of three values:

- `0` representing an empty cell.
- `1` representing a fresh orange.
- `2` representing a rotten orange.

Every minute, any fresh orange that is **4-directionally adjacent** to a rotten orange becomes rotten.

Return the minimum number of minutes that must elapse until no cell has a fresh orange. If this is impossible, return `-1`.

### Examples:

**Example 1:**

```
Input: grid = [[2,1,1],[1,1,0],[0,1,1]]
Output: 4
```

**Example 2:**

```
Input: grid = [[2,1,1],[0,1,1],[1,0,1]]
Output: -1
```

**Example 3:**

```
Input: grid = [[0,2]]
Output: 0
```

### Constraints:

- `1 <= m, n <= 10`
- `grid[i][j]` is `0`, `1`, or `2`.

### Implementation in Go

```go
func orangesRotting(grid [][]int) int {
    directions := [][]int{{1, 0}, {-1, 0}, {0, 1}, {0, -1}}
    queue := [][]int{}
    freshOranges := 0
    
    for i := 0; i < len(grid); i++ {
        for j := 0; j < len(grid[0]); j++ {
            if grid[i][j] == 2 {
                queue = append(queue, []int{i, j, 0})
            } else if grid[i][j] == 1 {
                freshOranges++
            }
        }
    }
    
    minutes := 0
    for len(queue) > 0 {
        r, c, time := queue[0][0], queue[0][1], queue[0][2]
        queue = queue[1:]
        
        for _, d := range directions {
            nr, nc := r+d[0], c+d[1]
            if nr >= 0 && nr < len(grid) && nc >= 0 && nc < len(grid[0]) && grid[nr][nc] == 1 {
                grid[nr][nc] = 2
                freshOranges--
                queue = append(queue, []int{nr, nc, time + 1})
                minutes = time + 1
            }
        }
    }
    
    if freshOranges > 0 {
        return -1
    }
    return minutes
}
```

### Implementation in Java

```java
import java.util.*;

class Solution {
    public int orangesRotting(int[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        Queue<int[]> queue = new LinkedList<>();
        int freshOranges = 0, minutes = 0;
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (grid[i][j] == 2) {
                    queue.offer(new int[]{i, j, 0});
                } else if (grid[i][j] == 1) {
                    freshOranges++;
                }
            }
        }
        
        int[][] directions = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int r = current[0], c = current[1], time = current[2];
            
            for (int[] d : directions) {
                int nr = r + d[0], nc = c + d[1];
                if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && grid[nr][nc] == 1) {
                    grid[nr][nc] = 2;
                    freshOranges--;
                    queue.offer(new int[]{nr, nc, time + 1});
                    minutes = time + 1;
                }
            }
        }
        
        return freshOranges == 0 ? minutes : -1;
    }
}
```


### Implementation in Python

```python
from collections import deque

class Solution:
    def orangesRotting(self, grid: list[list[int]]) -> int:
        rows, cols = len(grid), len(grid[0])
        queue = deque()
        fresh_oranges = 0
        minutes = 0

        for i in range(rows):
            for j in range(cols):
                if grid[i][j] == 2:
                    queue.append((i, j, 0))  
                elif grid[i][j] == 1:
                    fresh_oranges += 1

        directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        while queue:
            r, c, time = queue.popleft()

            for dr, dc in directions:
                nr, nc = r + dr, c + dc
                if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == 1:
                    grid[nr][nc] = 2
                    fresh_oranges -= 1
                    queue.append((nr, nc, time + 1))
                    minutes = time + 1

        return minutes if fresh_oranges == 0 else -1
```


## Conclusion

This problem can be efficiently solved using **Breadth-First Search (BFS)** to simulate the rotting process. The time complexity is approximately **O(m Ã— n)**, where `m` is the number of rows and `n` is the number of columns.
